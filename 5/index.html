គោលដៅ 5: សំរេចសមភាពយេនឌ័រនិងផ្តល់អំណាចដល់ស្ត្រីនិងក្មេងស្រីទាំងអស់

ចំណុចដៅនិងសូចនាករ

គោលដៅ 5.1

បញ្ចប់រាល់ទម្រង់នៃការរើសអើងប្រឆាំងនឹងស្ត្រីនិងក្មេងស្រីគ្រប់ទីកន្លែង

សូចនាករ 5.1.1

ក្របខ័ណ្ឌច្បាប់មានឬអត់ក្នុងការលើកកម្ពស់ពង្រឹងនិងត្រួតពិនិត្យសមភាពនិងការរើសអើងមិនឈរលើមូលដ្ឋាននៃការរួមភេទ ស្ថានភាពសូចនាករ បានរាយការណ៍

គោលដៅ 5.2

លុបបំបាត់រាល់ទម្រង់នៃអំពើហឹង្សាលើស្ត្រីនិងក្មេងស្រីទាំងអស់ក្នុងវិស័យសាធារណៈនិងឯកជនរួមទាំងការជួញដូរនិងការធ្វើអាជីវកម្មផ្លូវភេទនិងប្រភេទផ្សេងទៀតនៃការកេងប្រវ័ញ្ច។

សូចនាករ 5.2.1

សមាមាត្រនៃស្ត្រីនិងក្មេងស្រីដែលមានដៃគូដែលមិនធ្លាប់មានអាយុពី ១៥ ឆ្នាំឡើងទៅទទួលរងនូវអំពើហឹង្សាលើរាងកាយផ្លូវភេទឬផ្លូវចិត្តដោយដៃគូជិតស្និទ្ធឬបច្ចុប្បន្នក្នុង ១២ ខែមុនតាមទម្រង់នៃអំពើហិង្សានិងតាមអាយុ ស្ថានភាពសូចនាករ បានរាយការណ៍

សូចនាករ 5.2.2

សមាមាត្ររបស់ស្ត្រីនិងក្មេងស្រីដែលមានអាយុចាប់ពី ១៥ ឆ្នាំឡើងទៅទទួលរងនូវអំពើហឹង្សាផ្លូវភេទពីសំណាក់មនុស្សក្រៅពីដៃគូជិតស្និទ្ធក្នុងរយៈពេល ១២ ខែមុនតាមអាយុនិងទីកន្លែងដែលកើតមាន ស្ថានភាពសូចនាករ បានរាយការណ៍

គោលដៅ 5.3

លុបបំបាត់រាល់ការប្រព្រឹត្តដែលមានគ្រោះថ្នាក់ដូចជាការរៀបការការរៀបការដំបូងនិងការបង្ខំនិងការប្តូរប្រដាប់បន្តពូជស្ត្រី

សូចនាករ 5.3.1

សមាមាត្ររបស់ស្ត្រីដែលមានអាយុចាប់ពី ២០-២៤ ឆ្នាំដែលបានរៀបការឬនៅក្នុងសហជីពមុនអាយុ ១៥ ឆ្នាំនិងមុនអាយុ ១៨ ឆ្នាំ ស្ថានភាពសូចនាករ បានរាយការណ៍

គោលដៅ 5.4

ទទួលស្គាល់និងផ្តល់តម្លៃដល់ការថែទាំមិនយកប្រាក់និងការងារតាមផ្ទះតាមរយៈការផ្តល់សេវាសាធារណៈហេដ្ឋារចនាសម្ព័ន្ធនិងគោលនយោបាយគាំពារសង្គមនិងការលើកកម្ពស់ការទទួលខុសត្រូវរួមគ្នានៅក្នុងគ្រួសារនិងគ្រួសារតាមលក្ខណៈជាតិ។

សូចនាករ 5.4.1

សមាមាត្រនៃពេលវេលាដែលបានចំណាយទៅលើការងារថែទាំនិងថែទាំតាមផ្ទះដែលមិនបានបង់ដោយភេទអាយុនិងទីតាំង ស្ថានភាពសូចនាករ បានរាយការណ៍

គោលដៅ 5.5

ធានាឱ្យមានការចូលរួមពេញលេញនិងប្រកបដោយប្រសិទ្ធភាពរបស់ស្ត្រីនិងឱកាសស្មើគ្នាសម្រាប់ភាពជាអ្នកដឹកនាំនៅគ្រប់កម្រិតនៃការសម្រេចចិត្តក្នុងជីវិតនយោបាយសេដ្ឋកិច្ចនិងជីវិតសាធារណៈ

សូចនាករ 5.5.1

សមាមាត្រនៃអាសនៈដែលស្ត្រីកាន់កាប់នៅក្នុង (ក) រដ្ឋសភាជាតិនិង (ខ) រដ្ឋាភិបាលមូលដ្ឋាន ស្ថានភាពសូចនាករ បានរាយការណ៍

សូចនាករ 5.5.2

សមាមាត្រស្ត្រីក្នុងមុខតំណែងជាអ្នកគ្រប់គ្រង ស្ថានភាពសូចនាករ បានរាយការណ៍

សូចនាករ 5.5.3

គោលដៅ 5.6

ធានាឱ្យមានសិទ្ធិជាសកលចំពោះសិទ្ធិផ្លូវភេទនិងសុខភាពបន្តពូជនិងសិទ្ធិបន្តពូជដូចដែលបានព្រមព្រៀងគ្នាស្របតាមកម្មវិធីសកម្មភាពនៃសន្និសីទអន្តរជាតិស្តីពីចំនួនប្រជាជននិងការអភិវឌ្ឍនិងវេទិកាសកម្មភាពទីក្រុងប៉េកាំងនិងឯកសារលទ្ធផលនៃសន្និសីទត្រួតពិនិត្យរបស់ពួកគេ

សូចនាករ 5.6.1

សមាមាត្រនៃស្ត្រីដែលមានអាយុពី ១៥-៤៩ ឆ្នាំដែលធ្វើការសម្រេចចិត្តផ្ទាល់ខ្លួនទាក់ទងនឹងទំនាក់ទំនងផ្លូវភេទការប្រើប្រាស់ពន្យារកំណើតនិងការថែទាំសុខភាពបន្តពូជ ស្ថានភាពសូចនាករ បានរាយការណ៍

សូចនាករ 5.6.2

ចំនួនប្រទេសដែលមានច្បាប់និងបទប្បញ្ញត្តិដែលធានាសិទ្ធិពេញលេញនិងស្មើភាពគ្នាចំពោះស្ត្រីនិងបុរសដែលមានអាយុចាប់ពី ១៥ ឆ្នាំឡើងទៅទទួលបានការថែទាំសុខភាពផ្លូវភេទនិងបន្តពូជព័ត៌មាននិងការអប់រំ ស្ថានភាពសូចនាករ បានរាយការណ៍

គោលដៅ 5.c

អនុម័តនិងពង្រឹងគោលនយោបាយឱ្យបានត្រឹមត្រូវនិងច្បាប់ដែលអាចអនុវត្តបានសម្រាប់ការលើកកម្ពស់សមភាពយេនឌ័រនិងការផ្តល់សិទ្ធិអំណាចដល់ស្ត្រីនិងក្មេងស្រីគ្រប់ជាន់ថ្នាក់

សូចនាករ 5.c.1

សមាមាត្រនៃប្រទេសដែលមានប្រព័ន្ធដើម្បីតាមដាននិងធ្វើការបែងចែកជាសាធារណៈសម្រាប់សមភាពយេនឌ័រនិងការផ្តល់សិទ្ធិអំណាចដល់ស្ត្រី ស្ថានភាពសូចនាករ បានរាយការណ៍
